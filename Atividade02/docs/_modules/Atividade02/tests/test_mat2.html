        assert m[0,0] == 0
        assert m[0,1] == 0
        assert m[1,0] == 0
        assert m[1,1] == 0

        m[0,0] = 1
        m[0,1] = 2
        m[1,0] = 3
        m[1,1] = 4

        assert m[0,1] == 2
        assert m[1,0] == 3
        assert m[1,1] == 4

        assert np.all(m[1] == np.array([3, 4]))

        assert np.all(m[:,0] == np.array([1, 3]))
        assert np.all(m[:,1] == np.array([2, 4]))
