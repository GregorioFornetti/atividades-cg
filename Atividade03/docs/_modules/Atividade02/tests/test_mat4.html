        assert np.all(m[:,0] == np.array([1, 5, 9, 13]))
        assert np.all(m[:,1] == np.array([2, 6, 10, 14]))
        assert np.all(m[:,2] == np.array([3, 7, 11, 15]))
        assert np.all(m[:,3] == np.array([4, 8, 12, 16]))
